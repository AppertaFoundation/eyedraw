108 /**
109  * Draws doodle or performs a hit test if a Point parameter is passed
110  *
111  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
112  */
113 ED.Square.prototype.draw = function(_point)
114 {
115 	// Get context
116 	var ctx = this.drawing.context;
117 	
118 	// Call draw method in superclass
119 	ED.Square.superclass.draw.call(this, _point);
120     
121     // Set font
122     ctx.font = this.lableFont;
123     
124     // set lable text
125     this.lableText = "Hello";
126     
127     // Calculate pixel width of text with padding
128     this.lableWidth = ctx.measureText(this.lableText).width + this.padding * 2;
129 	
130 	// Boundary path
131 	ctx.beginPath();
132 	
133 	// Square
134 	ctx.rect(-50, -50, 100, 100);
135 	
136 	// Close path
137 	ctx.closePath();
138 	
139 	// Set line attributes
140 	ctx.lineWidth = 2;
141 	ctx.fillStyle = "green";
142 	ctx.strokeStyle = "blue";
143 	
144 	// Draw boundary path (also hit testing)
145 	this.drawBoundary(_point);
146 	
147 	// Other stuff here
148 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
149 	{
150 		ctx.beginPath();
151 		ctx.rect(-40, -20, 20, 20);
152 		ctx.lineWidth = 2;
153 		ctx.fillStyle = "red";
154 		ctx.strokeStyle = "blue";
155 		ctx.fill();
156 		ctx.stroke();
157         
158         // Draw text
159         ctx.fillText(this.lableText, -this.lableWidth/2 + this.padding, this.lableHeight/6);
160 	}
161 	
162 	// Coordinates of handles (in canvas plane)
163 	this.handleArray[0].location = this.transform.transformPoint(new ED.Point(-50, 50));
164 	this.handleArray[1].location = this.transform.transformPoint(new ED.Point(-50, -50));
165 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(50, -50));
166 	this.handleArray[3].location = this.transform.transformPoint(new ED.Point(50, 50));
167 	this.handleArray[4].location = this.transform.transformPoint(new ED.Point(this.apexX, this.apexY));
168 	
169 	// Draw handles if selected
170 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
171 	
172 	// Return value indicating successful hittest
173 	return this.isClicked;
174 }

219  *
220  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
221  */
222 ED.Fundus.prototype.draw = function(_point)
223 {
224 	// Get context
225 	var ctx = this.drawing.context;
226 	
227 	// Call draw method in superclass
228 	ED.Fundus.superclass.draw.call(this, _point);
229 
230 	// Boundary path
231 	ctx.beginPath();
232 	
233 	// Ora
234 	ctx.arc(0,0,480,0,Math.PI*2,true);
235 	
236 	// Close path
237 	ctx.closePath();
238 	
239 	// Set line attributes
240 	ctx.lineWidth = 2;
241 	this.isFilled = false;
242 	ctx.strokeStyle = "red";
243 	
244 	// Draw boundary path (also hit testing)
245 	this.drawBoundary(_point);
246 	
247 	// Other stuff here
248 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
249 	{
250 		// These values different for right and left side
251 		if(this.drawing.eye != ED.eye.Right)
252 		{
253 			var startX = 200;
254 			var midX = 100;
255 			var ctrlX = -50;
256 			var endX = -100;
257 			var foveaX = 100;
258 		}
259 		else
260 		{
261 			var startX = -200;
262 			var midX = -100;
263 			var ctrlX = 50;
264 			var endX = 100;
265 			var foveaX = -100;			
266 		}
267 		
268 		// Superior arcades
269 		ctx.moveTo(startX, -50);
270 		ctx.bezierCurveTo(midX, -166, 0, -62, 0, -12);
271 		ctx.bezierCurveTo(0, -40, ctrlX, -100, endX, -50);
272 		
273 		// Inferior arcades
274 		ctx.moveTo(startX, 50);
275 		ctx.bezierCurveTo(midX, 166, 0, 62, 0, 12);
276 		ctx.bezierCurveTo(0, 40, ctrlX, 100, endX, 50);
277 		
278 		// Small cross marking fovea
279 		var crossLength = 10;
280 		ctx.moveTo(foveaX, -crossLength);
281 		ctx.lineTo(foveaX, crossLength);
282 		ctx.moveTo(foveaX - crossLength, 0);
283 		ctx.lineTo(foveaX + crossLength, 0);
284 		
285 		// Optic disk and cup 
286 		ctx.moveTo(25, 0);
287 		ctx.arc(0,0,25,0,Math.PI*2,true);
288 		ctx.moveTo(12, 0);
289 		ctx.arc(0,0,12,0,Math.PI*2,true);
290 		
291 		// Draw it
292 		ctx.stroke();
293 	}
294 	
295 	// Return value indicating successful hittest
296 	return this.isClicked;
297 }

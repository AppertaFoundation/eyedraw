117         // Upper Eye lid

128 	// Return value indicating successful hittest

194 /**
195  * Draws doodle or performs a hit test if a Point parameter is passed
196  *
197  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
198  */
199 ED.Shading.prototype.draw = function(_point)
200 {
201 	// Get context
202 	var ctx = this.drawing.context;
203 	
204 	// Call draw method in superclass
205 	ED.Shading.superclass.draw.call(this, _point);
206 
207 	// Boundary path
208 	ctx.beginPath();
209     
210 	// Rectangular area
211 	ctx.rect(-300, -100, 600, 200);
212     
213 	// Close path
214 	ctx.closePath();
215     
216     // create pattern
217 	ctx.fillStyle = "rgba(190, 190, 190, 0.55)";
218 	ctx.strokeStyle = "rgba(0, 0, 0, 0)";
219 	
220 	// Draw boundary path (also hit testing)
221 	this.drawBoundary(_point);
222 	
223 	// Other stuff here
224 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
225 	{
226         ctx.beginPath();
227         ctx.moveTo(-300, 100);
228         var dash = 42;
229         var gap = 20;
230         var length = 0;
231         while (length < 540)
232         {
233             length += dash;
234             ctx.lineTo(-300 + length, 100);
235             length += gap;            
236             ctx.moveTo(-300 + length, 100);
237         }
238         ctx.lineTo(300, 100);
239         
240         // Draw line
241         ctx.lineWidth = 12;
242         ctx.strokeStyle = "rgba(80, 80, 80, 1)";
243         ctx.stroke();
244 	}
245 	
246 	// Coordinates of handles (in canvas plane)
247 	this.handleArray[3].location = this.transform.transformPoint(new ED.Point(300, 100));
248 	
249 	// Draw handles if selected
250 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
251 	
252 	// Return value indicating successful hittest
253 	return this.isClicked;
254 }

318 /**
319  * Draws doodle or performs a hit test if a Point parameter is passed
320  *
321  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
322  */
323 ED.UpShoot.prototype.draw = function(_point)
324 {
325     // Use scale to flip arrow into correct position for quadrant
326     this.scaleX = this.originX/Math.abs(this.originX);
327     this.scaleY = this.originY/Math.abs(this.originY);
328     
329 	// Get context
330 	var ctx = this.drawing.context;
331 	
332 	// Call draw method in superclass
333 	ED.UpShoot.superclass.draw.call(this, _point);
334     
335 	// Boundary path
336 	ctx.beginPath();
337     
338 	// Rectangular area
339 	ctx.rect(-100, -100, 200, 200);
340     
341 	// Close path
342 	ctx.closePath();
343 	
344 	// Set line attributes
345 	ctx.lineWidth = 0;
346 	ctx.fillStyle = "rgba(0, 0, 0, 0)";
347 	ctx.strokeStyle = ctx.fillStyle;
348     	
349 	// Draw boundary path (also hit testing)
350 	this.drawBoundary(_point);
351 	
352 	// Other stuff here
353 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
354 	{
355         // Arrow shaft
356         ctx.beginPath();
357         ctx.moveTo(-100, -100);
358         ctx.lineTo(100, -100);
359         ctx.lineTo(100, 80);
360         
361         ctx.lineWidth = 6;
362         ctx.lineJoin = 'miter';
363         ctx.strokeStyle = "rgba(80, 80, 80, 1)";
364         ctx.stroke();
365         
366         // Arrow head
367         ctx.beginPath();
368         ctx.moveTo(100, 100);
369         ctx.lineTo(80, 70);
370         ctx.lineTo(120, 70);
371         ctx.closePath();
372         
373         ctx.fillStyle = "rgba(80, 80, 80, 1)";
374         ctx.fill();
375 	}
376 	
377 	// Draw handles if selected
378 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
379 	
380 	// Return value indicating successful hittest
381 	return this.isClicked;
382 }

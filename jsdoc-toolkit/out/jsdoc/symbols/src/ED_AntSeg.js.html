102 /**
103  * Draws doodle or performs a hit test if a Point parameter is passed
104  *
105  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
106  */
107 ED.AntSeg.prototype.draw = function(_point)
108 {
109 	// Get context
110 	var ctx = this.drawing.context;
111 	
112 	// Call draw method in superclass
113 	ED.AntSeg.superclass.draw.call(this, _point);
114     
115 	// Radius of limbus
116 	var ro = 380;
117     var ri = -this.apexY;
118     //var r = ri + (ro - ri)/2;
119 	
120 	// Calculate parameters for arcs
121 	var arcStart = 0;
122 	var arcEnd = 2 * Math.PI;
123     
124 	// Boundary path
125 	ctx.beginPath();
126     
127 	// Do a 360 arc
128 	ctx.arc(0, 0, ro, arcStart, arcEnd, true);
129     
130     // Move to inner circle
131     ctx.moveTo(ri, 0);
132     
133 	// Arc back the other way
134 	ctx.arc(0, 0, ri, arcEnd, arcStart, false);
135 	
136 	// Set line attributes
137 	ctx.lineWidth = 4;	
138 	ctx.fillStyle = "rgba(100, 200, 250, 0.75)";
139 	ctx.strokeStyle = "gray";
140 	
141 	// Draw boundary path (also hit testing)
142 	this.drawBoundary(_point);
143 	
144 	// Other paths and drawing here
145 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
146 	{
147         // Pseudo exfoliation
148         if (this.hasPXE)
149         {
150             ctx.lineWidth = 8;
151             ctx.strokeStyle = "darkgray";
152             
153             var rl = ri * 0.8;
154             var rp = ri * 1.05;
155             var segments = 36;
156             var i;
157             var phi = Math.PI * 2/segments;
158             
159             // Loop around alternating segments
160             for (i = 0; i < segments; i++)
161             {
162                 // PXE on lens
163                 ctx.beginPath();
164                 ctx.arc(0, 0, rl, i * phi, i * phi + phi/2, false);
165                 ctx.stroke();
166                 
167                 // PXE on pupil
168                 ctx.beginPath();
169                 ctx.arc(0, 0, rp, i * phi, i * phi + phi/2, false);
170                 ctx.stroke();
171             }
172         }
173 	}
174     
175 	// Coordinates of handles (in canvas plane)
176 	this.handleArray[4].location = this.transform.transformPoint(new ED.Point(this.apexX, this.apexY));
177 	
178 	// Draw handles if selected
179 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
180     
181 	// Return value indicating successful hit test
182 	return this.isClicked;
183 }

511 ED.CorticalCataract.prototype.draw = function(_point)

514 	var ctx = this.drawing.context;

516 	// Call draw method in superclass
517 	ED.CorticalCataract.superclass.draw.call(this, _point);
518 	
519 	// Boundary path
520 	ctx.beginPath();
521 
522 	// CorticalCataract
523     ctx.arc(0, 0, 240, 0, Math.PI * 2, true);
524 	
525 	// Close path
526 	ctx.closePath();
527 	
528 	// Set line attributes
529 	ctx.lineWidth = 1;
530     
531 	ctx.fillStyle = "rgba(0,0,0,0)";
532 	ctx.strokeStyle = "gray";
533 	
534 	// Draw boundary path (also hit testing)
535 	this.drawBoundary(_point);
536 	
537 	// Other stuff here
538 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
539 	{
540         // Parameters
541         var n = 16;                 // Number of cortical spokes
542         var ro = 240;               // Outer radius of cataract
543         var theta = 2 * Math.PI/n;	// Angle of outer arc of cortical shard
544         var phi = theta/2;          // Half theta
545         var ri = -this.apexY;       // Radius of inner clear area
546         ctx.fillStyle = "rgba(200,200,200,0.75)";
547         
548         // Draw cortical spokes
549         var i;
550         for (i = 0; i < n; i++)
551         {
552             ctx.beginPath();
553             var startAngle = i * theta - phi - Math.PI/2;
554             var endAngle = startAngle + theta;
555             ctx.arc(0, 0, ro, startAngle, endAngle, false);
556             var p = new ED.Point(0, 0);
557             p.setWithPolars(ri, i * theta); 
558             ctx.lineTo(p.x, p.y);
559             ctx.closePath();
560             ctx.fill()
561         }
562 	}
563 	
564 	// Coordinates of handles (in canvas plane)
565 	this.handleArray[4].location = this.transform.transformPoint(new ED.Point(this.apexX, this.apexY));
566 	
567 	// Draw handles if selected
568 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
569 	
570 	// Return value indicating successful hittest
571 	return this.isClicked;

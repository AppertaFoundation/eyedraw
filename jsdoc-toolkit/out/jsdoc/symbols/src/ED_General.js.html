189 /**
190  * Draws doodle or performs a hit test if a Point parameter is passed
191  *
192  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
193  */
194 ED.Freehand.prototype.draw = function(_point)
195 {
196 	// Get context
197 	var ctx = this.drawing.context;
198 	
199 	// Call draw method in superclass
200 	ED.Freehand.superclass.draw.call(this, _point);
201 	
202 	// Boundary path
203 	ctx.beginPath();
204 	
205 	// Freehand
206 	ctx.rect(-150, -150, 300, 300);
207 	
208 	// Close path
209 	ctx.closePath();
210 	
211 	// Set line attributes
212 	ctx.lineWidth = 2;
213     this.isFilled = false;
214     ctx.strokeStyle = "rgba(0, 0, 0, 0)";
215     if (this.isSelected) ctx.strokeStyle = "gray";
216     if (this.isForDrawing) ctx.strokeStyle = "blue";
217 
218 	// Draw boundary path (also hit testing)
219 	this.drawBoundary(_point);
220 	
221 	// Other stuff here
222 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
223 	{
224         // Iterate through squiggles, drawing them
225         for (var i = 0; i < this.squiggleArray.length; i++)
226         {
227             var squiggle = this.squiggleArray[i];
228             
229             ctx.beginPath();
230             
231             // Squiggle attributes
232             ctx.lineWidth = squiggle.thickness;
233             ctx.strokeStyle = squiggle.colour.rgba();
234             ctx.fillStyle = squiggle.colour.rgba();
235             
236             // Iterate through squiggle points
237             for (var j = 0; j < squiggle.pointsArray.length; j++)
238             {
239                 ctx.lineTo(squiggle.pointsArray[j].x, squiggle.pointsArray[j].y);
240             }
241             
242             // Draw squiggle
243             ctx.stroke();
244             
245             // Optionally fill if squiggle is complete (stops filling while drawing)
246             if (squiggle.filled && squiggle.complete) ctx.fill();
247         }
248 	}
249 	
250 	// Coordinates of handles (in canvas plane)
251 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(150, -150));
252 	
253 	// Draw handles if selected but not if for drawing
254 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
255 	
256 	// Return value indicating successful hittest
257 	return this.isClicked;
258 }

341 /**
342  * Draws doodle or performs a hit test if a Point parameter is passed
343  *
344  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
345  */
346 ED.Lable.prototype.draw = function(_point)
347 {
348 	// Get context
349 	var ctx = this.drawing.context;
350 	
351 	// Call draw method in superclass
352 	ED.Lable.superclass.draw.call(this, _point);
353     
354     // Set font
355     ctx.font = this.lableFont;
356     
357     // Calculate pixel width of text with padding
358     this.lableWidth = ctx.measureText(this.lableText).width + this.padding * 2;
359 	
360 	// Boundary path
361 	ctx.beginPath();
362 	
363 	// Lable boundary
364 	ctx.rect(-this.lableWidth/2, -this.lableHeight/2, this.lableWidth, this.lableHeight);
365     
366 	// Close path
367 	ctx.closePath();
368 	
369 	// Set line attributes
370 	ctx.lineWidth = 2;
371     this.isFilled = false;
372     ctx.strokeStyle = "rgba(0, 0, 0, 0)";
373     if (this.isSelected) ctx.strokeStyle = "gray";
374     
375 	// Draw boundary path (also hit testing)
376 	this.drawBoundary(_point);
377 	
378 	// Other stuff here
379 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
380 	{    
381         // Draw boundary
382         ctx.stroke();
383         
384         // Draw text
385         ctx.fillText(this.lableText, -this.lableWidth/2 + this.padding, this.lableHeight/6);
386     }
387 	
388 	// Coordinates of handles (in canvas plane)
389 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(this.lableWidth/2, -this.lableHeight/2));
390 	
391 	// Draw handles if selected but not if for drawing
392 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
393 	
394 	// Return value indicating successful hittest
395 	return this.isClicked;
396 }

636     
637 	// Set line attributes

639

737 /**
738  * Draws doodle or performs a hit test if a Point parameter is passed
739  *
740  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
741  */
742 ED.PointInLine.prototype.draw = function(_point)
743 {
744 	// Get context
745 	var ctx = this.drawing.context;
746 	
747 	// Call draw method in superclass
748 	ED.PointInLine.superclass.draw.call(this, _point);
749 	
750 	// Boundary path
751 	ctx.beginPath();
752 	
753 	// Round hole
754 	ctx.arc(0,0,8,0,Math.PI*2,true);
755     
756 	// Close path
757 	ctx.closePath();
758 	
759 	// Set line attributes
760 	ctx.lineWidth = 4;
761 	ctx.fillStyle =  "rgba(20,20,20,1)";
762 	ctx.strokeStyle = "rgba(20,20,20,1)";
763 	
764 	// Draw boundary path (also hit testing)
765 	this.drawBoundary(_point);
766 	
767 	// Other stuff here
768 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
769 	{
770 	}
771 	
772 	// Return value indicating successful hittest
773 	return this.isClicked;
774 }

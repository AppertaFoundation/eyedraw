 97 /**
 98  * Draws doodle or performs a hit test if a Point parameter is passed
 99  *
100  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
101  */
102 ED.Breasts.prototype.draw = function(_point)
103 {
104 	// Get context
105 	var ctx = this.drawing.context;
106 	
107 	// Call draw method in superclass
108 	ED.Breasts.superclass.draw.call(this, _point);
109 	
110 	// Boundary path
111 	ctx.beginPath();
112 
113     ctx.moveTo(-400,-200);
114     ctx.bezierCurveTo(-400,-160,-300,-150,-300,-100);
115     ctx.bezierCurveTo(-300,-50,-450,20,-450,150);
116     ctx.bezierCurveTo(-450,200,-400,300,-300,300);
117     ctx.bezierCurveTo(-200,300,-100,200,0,200);
118     ctx.bezierCurveTo(100,200,200,300,300,300);
119     ctx.bezierCurveTo(400,300,450,200,450,150);
120     ctx.bezierCurveTo(450,20,300,-50,300,-100);
121     ctx.bezierCurveTo(300,-150,400,-160,400,-200);
122     
123 	// Close path
124 	//ctx.closePath();
125 	
126 	// Set line attributes
127 	ctx.lineWidth = 8;
128 	ctx.fillStyle = "rgba(0, 0, 0, 0)";
129 	ctx.strokeStyle = "gray";
130 	
131 	// Draw boundary path (also hit testing)
132 	this.drawBoundary(_point);
133 	
134 	// Other stuff here
135 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
136 	{
137         // Areola
138         ctx.beginPath();
139         ctx.arc(300,150,40,0,Math.PI*2,true);
140         ctx.arc(-300,150,40,0,Math.PI*2,true);
141         ctx.fillStyle = "pink";
142         ctx.fill();
143         // Nipple
144         ctx.beginPath();
145         ctx.arc(300,150,20,0,Math.PI*2,true);
146         ctx.arc(-300,150,20,0,Math.PI*2,true);
147         ctx.fillStyle = "brown";
148         ctx.fill();
149 	}
150 	
151 	// Coordinates of handles (in canvas plane)
152     this.handleArray[1].location = this.transform.transformPoint(new ED.Point(-400, -200));
153 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(400, -200));
154 	
155 	// Draw handles if selected
156 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
157 	
158 	// Return value indicating successful hittest
159 	return this.isClicked;
160 }

282 	ctx.closePath();
283 	
284 	// Set line attributes
285 	ctx.lineWidth = 0;

295         ctx.beginPath();
296         ctx.lineWidth = 8;
297         ctx.strokeStyle = "brown";
298         ctx.moveTo(-100,0);
299         ctx.lineTo(100,0);
300         ctx.moveTo(-50,-50);
301         ctx.lineTo(-50,50);        
302         ctx.stroke();
303 	}
304 	
305 	// Coordinates of handles (in canvas plane)
306 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(100, -50));
307 	
308 	// Draw handles if selected

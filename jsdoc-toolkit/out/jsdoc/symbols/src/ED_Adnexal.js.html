 31 /**
 32  * Anterior segment with adjustable sized pupil
 33  *
 34  * @class AdnexalEye
 35  * @property {String} className Name of doodle subclass
 36  * @param {Drawing} _drawing
 37  * @param {Int} _originX
 38  * @param {Int} _originY
 39  * @param {Int} _apexX
 40  * @param {Int} _apexY
 41  * @param {Float} _scaleX
 42  * @param {Float} _scaleY
 43  * @param {Float} _arc
 44  * @param {Float} _rotation
 45  * @param {Int} _order
 46  */
 47 ED.AdnexalEye = function(_drawing, _originX, _originY, _apexX, _apexY, _scaleX, _scaleY, _arc, _rotation, _order)
 48 {
 49 	// Call superclass constructor
 50 	ED.Doodle.call(this, _drawing, _originX, _originY, _apexX, _apexY, _scaleX, _scaleY, _arc, _rotation, _order);
 51 	
 52 	// Set classname
 53 	this.className = "AdnexalEye";
 54 }

 97 /**
 98  * Draws doodle or performs a hit test if a Point parameter is passed
 99  *
100  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
101  */
102 ED.AdnexalEye.prototype.draw = function(_point)
103 {
104 	// Get context
105 	var ctx = this.drawing.context;
106 	
107 	// Call draw method in superclass
108 	ED.AdnexalEye.superclass.draw.call(this, _point);
109    
110 	// Boundary path
111 	ctx.beginPath();
112     
113 	// Draw outline of eyelids
114     var s = 1;
115     ctx.moveTo(-400* s,0* s);
116     ctx.bezierCurveTo(-400* s,-50* s,-200* s,-250* s,0* s,-250* s);
117     ctx.bezierCurveTo(200* s,-250* s,300* s,-100* s,350* s,-50* s);
118     ctx.bezierCurveTo(400* s,-60* s,440* s,-10* s,440* s,0* s);
119     ctx.bezierCurveTo(440* s,10* s,400* s,50* s,350* s,50* s);
120     ctx.bezierCurveTo(300* s,100* s,200* s,250* s,0* s,250* s);
121     ctx.bezierCurveTo(-200* s,250* s,-400* s,50* s,-400* s,0* s);
122 
123     // Close path
124     //ctx.closePath();
125 	
126 	// Set line attributes
127 	ctx.lineWidth = 4;	
128 	ctx.fillStyle = "rgba(100, 200, 250, 0)";
129 	ctx.strokeStyle = "gray";
130 	
131 	// Draw boundary path (also hit testing)
132 	this.drawBoundary(_point);
133 	
134 	// Other paths and drawing here
135 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
136 	{
137         // Boundary path
138         ctx.beginPath();
139         
140         // Do a 360 arc
141         ctx.arc(0, 0, 190 * s, 0, 2 * Math.PI, true);
142         
143         // Move to inner circle
144         ctx.moveTo(120 * s, 0);
145         
146         // Arc back the other way
147         ctx.arc(0, 0, 120 * s,  2 * Math.PI, 0, false);
148         
149         // Set line attributes
150         ctx.lineWidth = 4;	
151         ctx.fillStyle = "rgba(100, 200, 250, 0.75)";
152         ctx.strokeStyle = "gray";
153         
154         ctx.fill();
155         ctx.stroke();
156 	}
157     
158 	// Coordinates of handles (in canvas plane)
159 	//this.handleArray[4].location = this.transform.transformPoint(new ED.Point(this.apexX, this.apexY));
160 	
161 	// Draw handles if selected
162 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
163     
164 	// Return value indicating successful hit test
165 	return this.isClicked;
166 }

168 /**
169  * Returns a string containing a text description of the doodle
170  *
171  * @returns {String} Description of doodle
172  */
173 ED.AdnexalEye.prototype.description = function()
174 {
175     var returnString = this.getGrade() + " pupil";
176     
177     if (this.hasPXE) returnString += " with pseudoexfoliation";
178 	
179 	return returnString;
180 }

100 /*
101  * Chris Raettig's function for getting accurate mouse position in all browsers
102  *
103  * @param {Object} obj Object to get offset for, usually canvas object
104  * @returns {Object} x and y values of offset
105  */
106 ED.findOffset = function(obj)
107 {
108     var curleft = curtop = 0;
109     if (obj.offsetParent) {
110         do {
111             curleft += obj.offsetLeft;
112             curtop += obj.offsetTop;
113         } while (obj = obj.offsetParent);
114         return { x: curleft, y: curtop };
115     }
116 }

139 /**
140  * A Drawing consists of one canvas element displaying one or more doodles;
141  * Doodles are drawn in the 'doodle plane' consisting of a 1001 pixel square grid -500 to 500) with central origin, and negative Y upwards;
142  * Affine transforms are used to convert points in the doodle plane to the canvas plane, the plane of the canvas element;
143  * Each doodle contains additional transforms to handle individual position, rotation, and scale.
144  * 
145  * @class Drawing
146  * @property {Canvas} canvas A canvas element used to edit and display the drawing
147  * @property {Eye} eye Right or left eye (some doodles display differently according to side)
148  * @property {Bool} isEditable Flag indicating whether canvas is editable or not
149  * @property {Context} context The 2d context of the canvas element
150  * @property {Array} doodleArray Array of doodles in the drawing
151  * @property {AffineTransform} transform Transform converts doodle plane -> canvas plane
152  * @property {AffineTransform} inverseTransform Inverse transform converts canvas plane -> doodle plane
153  * @property {Doodle} selectedDoodle The currently selected doodle, null if no selection
154  * @property {Bool} mouseDown Flag indicating whether mouse is down in canvas
155  * @property {Mode} mode The current mouse dragging mode
156  * @property {Point} lastMousePosition Last position of mouse in canvas coordinates
157  * @property {Image} image Optional background image
158  * @property {Int} doubleClickMilliSeconds Duration of double click
159  * @property {Bool} newPointOnClick Flag indicating whether a mouse click will create a new PointInLine doodle
160  * @property {Bool} completeLine Flag indicating whether to draw an additional line to the first PointInLine doodle
161  * @property {Float} scale Scaling of transformation from canvas to doodle planes, preserving aspect ratio and maximising doodle plnae
162  * @property {Float} globalScaleFactor Factor used to scale all added doodles to this drawing, defaults to 1
163  * @param {Canvas} _canvas Canvas element 
164  * @param {Eye} _eye Right or left eye
165  * @param {String} _IDSuffix String suffix to identify HTML elements related to this drawing
166  * @param {Bool} _isEditable Flag indicating whether canvas is editable or not
167  */

241         var drawing = this;
242         
243         // Mouse listeners

258                                 var point = new ED.Point(e.pageX-offset.x,e.pageY-offset.y);

281                                 var point = new ED.Point(e.targetTouches[0].pageX - this.offsetLeft,e.targetTouches[0].pageY - this.offsetTop);

287                                 if (document.activeElement && document.activeElement.tagName == 'CANVAS') drawing.keydown(e);

318 /**
319  * Checks all images are loaded then calls onLoaded function to proceed with other processing
320  */
321 ED.Drawing.prototype.checkAllLoaded = function()
322 {
323     // Set flag to check loading
324     var allLoaded = true;
325     
326     // Iterate through array loading each image, checking all are loaded
327     for (var key in this.imageArray)
328     {
329         var imageLoaded = false;
330         if (this.imageArray[key].width > 0) imageLoaded = true;
331         
332         // Check all are loaded
333         allLoaded = allLoaded && imageLoaded;
334     }
335     
336     // If all are loaded, proceed with onLoaded event (if defined)
337     if (allLoaded && typeof(this.onLoaded) == 'function')
338     {
339         if (!this.onLoadedHasRun)
340         {
341             this.onLoaded();
342             this.onLoadedHasRun = true;
343         }
344     }
345 }

365 /**
366  * Loads doodles from passed set in JSON format into doodleArray
367  *
368  * @param {Set} _doodleSet Set of doodles from server
369  */
370 ED.Drawing.prototype.load = function(_doodleSet)
371 {
372 	// Iterate through set of doodles and load into doodle array
373 	for (var i = 0; i < _doodleSet.length; i++)
374 	{
375 		// Instantiate a new doodle object with parameters from doodle set
376 		this.doodleArray[i] = new ED[_doodleSet[i].subclass]
377 		(
378 			this,
379 			_doodleSet[i].originX,
380 			_doodleSet[i].originY,
381 			_doodleSet[i].radius,
382 			_doodleSet[i].apexX,
383 			_doodleSet[i].apexY,
384 			_doodleSet[i].scaleX,
385 			_doodleSet[i].scaleY,
386 			_doodleSet[i].arc,
387 			_doodleSet[i].rotation,
388 			_doodleSet[i].order
389 		);
390         
391         // Squiggle array
392         if (typeof(_doodleSet[i].squiggleArray) != 'undefined')
393         {
394             for (var j = 0; j < _doodleSet[i].squiggleArray.length; j++)
395             {
396                 // Get paramters and create squiggle
397                 var c = _doodleSet[i].squiggleArray[j].colour;
398                 var colour = new ED.Colour(c.red, c.green, c.blue, c.alpha);
399                 var thickness = _doodleSet[i].squiggleArray[j].thickness;
400                 var filled = _doodleSet[i].squiggleArray[j].filled;
401                 var squiggle = new ED.Squiggle(this.doodleArray[i], colour, thickness, filled);
402                 
403                 // Add points to squiggle and complete it
404                 var pointsArray = _doodleSet[i].squiggleArray[j].pointsArray;
405                 for (var k = 0; k < pointsArray.length; k++)
406                 {
407                     var point = new ED.Point(pointsArray[k].x, pointsArray[k].y);
408                     squiggle.addPoint(point);
409                 }
410                 squiggle.complete = true;
411                 
412                 // Add squiggle to doodle's squiggle array
413                 this.doodleArray[i].squiggleArray.push(squiggle);
414             }
415         }
416 	}
417 	
418 	// Sort array by order (puts back doodle first)
419 	this.doodleArray.sort(function(a,b){return a.order - b.order});
420 }

454 /**
455  * Draws all doodles for this drawing
456  */ 
457 ED.Drawing.prototype.drawAllDoodles = function()
458 {
459     // Draw any connecting lines
460 
461     var ctx = this.context;
462     ctx.beginPath();
463     var started = false;
464     var startPoint;
465     
466     for (var i = 0; i < this.doodleArray.length; i++)
467     {
468         if (this.doodleArray[i].isPointInLine)
469         {
470             // Start or draw line
471             if (!started)
472             {
473                 ctx.moveTo(this.doodleArray[i].originX, this.doodleArray[i].originY);
474                 started = true;
475                 startPoint = new ED.Point(this.doodleArray[i].originX, this.doodleArray[i].originY);
476             }
477             else
478             {
479                 ctx.lineTo(this.doodleArray[i].originX, this.doodleArray[i].originY);
480             }
481         }
482     }
483     
484     // Optionally add line to start
485     if (this.completeLine && typeof(startPoint) != 'undefined')
486     {
487         ctx.lineTo(startPoint.x, startPoint.y);
488     }
489     
490     // Draw lines
491     if (started)
492     {
493         ctx.lineWidth = 4;
494         ctx.strokeStyle = "rgba(20,20,20,1)";
495         ctx.stroke();
496     }
497 
498     
499 	// Draw doodles
500 	for (var i = 0; i < this.doodleArray.length; i++)
501 	{
502 		// Save context (draw method of each doodle may alter it)
503 		this.context.save();
504 		
505 		// Draw doodle
506 		this.doodleArray[i].draw();
507 		
508 		// Restore context
509 		this.context.restore();
510 	}
511 }

514 /**
515  * Responds to mouse down event in canvas, cycles through doodles from front to back.
516  * Selected doodle is first selectable doodle to have click within boundary path.
517  * Double clicking on a selected doodle promotes it to drawing mode (if is drawable)
518  *
519  * @event
520  * @param {Point} _point Coordinates of mouse in canvas plane
521  */  
522 ED.Drawing.prototype.mousedown = function(_point)
523 {
524 	// Set flag to indicate dragging can now take place
525 	this.mouseDown = true;
526     
527     // Detect double click
528     if (ED.recentClick) this.doubleClick = true;
529     ED.recentClick = true;
530     var t = setTimeout("ED.recentClick = false;",this.doubleClickMilliSeconds);
531     
532 	// Set flag to indicate success
533 	var found = false;
534 	this.selectedDoodle = null;
535 
536 	// Cycle through doodles from front to back doing hit test
537 	for (var i = this.doodleArray.length - 1; i > -1; i--)
538 	{
539 		if (!found)
540 		{
541 			// Save context (draw method of each doodle may alter it)
542 			this.context.save();
543 		
544 			// Successful hit test?
545 			if (this.doodleArray[i].draw(_point))
546 			{
547 				if (this.doodleArray[i].isSelectable)
548 				{
549                     // If double clicked, go into drawing mode
550                     if (this.doubleClick && this.doodleArray[i].isSelected && this.doodleArray[i].isDrawable)
551                     {
552                         this.doodleArray[i].isForDrawing = true;
553                     }
554                     
555 					this.doodleArray[i].isSelected = true;
556 					this.selectedDoodle = this.doodleArray[i];
557 					found = true;
558                     
559                     // If for drawing, mouse down starts a new squiggle
560                     if (!this.doubleClick && this.doodleArray[i].isForDrawing)
561                     {
562                         // Add new squiggle
563                         this.doodleArray[i].addSquiggle();
564                     }
565 				}
566 			}
567 			// Ensure that unselected doodles are marked as such
568 			else
569 			{
570 				this.doodleArray[i].isSelected = false;
571                 this.doodleArray[i].isForDrawing = false;
572 			}
573 			
574 			// Restore context
575 			this.context.restore();
576 		}
577 		else
578 		{
579 			this.doodleArray[i].isSelected = false;
580             this.doodleArray[i].isForDrawing = false;
581 		}
582 		
583 		// Ensure drag flagged is off for each doodle
584 		this.doodleArray[i].isBeingDragged = false;
585 	}
586     
587     
588     if (this.newPointOnClick && !found)
589     {
590         var mousePosDoodlePlane = this.inverseTransform.transformPoint(_point);
591         
592         var newPointInLine = this.addDoodle('PointInLine');
593         newPointInLine.originX = mousePosDoodlePlane.x;
594         newPointInLine.originY = mousePosDoodlePlane.y;
595     }
596 	
597 	// Repaint
598 	this.repaint();
599 }

601 /**
602  * Responds to mouse move event in canvas according to the drawing mode
603  *
604  * @event
605  * @param {Point} _point coordinates of mouse in canvas plane
606  */
607 ED.Drawing.prototype.mousemove = function(_point)
608 {
609 	// Only drag if mouse already down and a doodle selected
610 	if (this.mouseDown && this.selectedDoodle != null)
611 	{
612 		// Dragging not started
613 		if (!this.selectedDoodle.isBeingDragged)
614 		{
615 			// Flag start of dragging manoeuvre
616 			this.selectedDoodle.isBeingDragged = true;
617 		}
618 		// Dragging in progress
619 		else
620 		{
621 			// Get mouse position in doodle plane
622 			var mousePosDoodlePlane = this.inverseTransform.transformPoint(_point);
623 			var lastMousePosDoodlePlane = this.inverseTransform.transformPoint(this.lastMousePosition);
624 			
625 			// Get mouse positions in selected doodle's plane
626 			var mousePosSelectedDoodlePlane = this.selectedDoodle.inverseTransform.transformPoint(_point);
627 			var lastMousePosSelectedDoodlePlane = this.selectedDoodle.inverseTransform.transformPoint(this.lastMousePosition);
628 			
629 			// Get mouse positions in canvas plane relative to centre
630 			var mousePosRelCanvasCentre = new ED.Point(_point.x - this.canvas.width/2, _point.y - this.canvas.height/2);
631 			var lastMousePosRelCanvasCentre = new ED.Point(this.lastMousePosition.x - this.canvas.width/2, this.lastMousePosition.y - this.canvas.height/2);
632 				
633 			// Get position of centre of display (canvas plane relative to centre) and of an arbitrary point vertically above
634 			var canvasCentre = new ED.Point(0, 0);
635 			var canvasTop = new ED.Point(0, -100);
636 			
637 			// Get coordinates of origin of doodle in doodle plane
638 			var doodleOrigin = new ED.Point(this.selectedDoodle.originX, this.selectedDoodle.originY);
639 
640 			// Get position of point vertically above doodle origin in doodle plane
641 			var doodleTop = new ED.Point(this.selectedDoodle.originX, this.selectedDoodle.originY - 100);
642 			
643 			// Effect of dragging depends on mode
644 			switch (this.mode)
645 			{
646 				case ED.Mode.None:
647 					break;
648 				case ED.Mode.Move:
649 					// If isMoveable is true, move doodle
650 					if (this.selectedDoodle.isMoveable)
651 					{
652                         // Enforce snap to quadrant
653                         if (this.selectedDoodle.snapToQuadrant)
654                         {
655                             if (mousePosDoodlePlane.x != 0)
656                             {
657                                 this.selectedDoodle.originX = this.selectedDoodle.quadrantPoint.x * mousePosDoodlePlane.x/Math.abs(mousePosDoodlePlane.x);
658                             }
659                             if (mousePosDoodlePlane.y != 0)
660                             {
661                                 this.selectedDoodle.originY = this.selectedDoodle.quadrantPoint.y * mousePosDoodlePlane.y/Math.abs(mousePosDoodlePlane.y);
662                             }
663                             
664                             // Doodle's move method sets orientation, pass pass origin deltas as zero, since explicity set above
665                             this.selectedDoodle.move(0,0);
666                         }
667                         else
668                         {

674                         }
675 					}
676 					// Otherwise rotate it (if isRotatable)
677 					else 
678 					{
679 						if (this.selectedDoodle.isRotatable)
680 						{
681 							// Calculate angles from centre to mouse positions relative to north
682 							var oldAngle = this.innerAngle(canvasTop, canvasCentre, lastMousePosRelCanvasCentre);
683 							var newAngle = this.innerAngle(canvasTop, canvasCentre, mousePosRelCanvasCentre);
684 							
685 							// Work out difference, and change doodle's angle of rotation by this amount
686 							var deltaAngle = newAngle - oldAngle;
687 							this.selectedDoodle.rotation += deltaAngle;
688                             
689                             // Adjust radius property
690                             var oldRadius = Math.sqrt(lastMousePosDoodlePlane.x * lastMousePosDoodlePlane.x + lastMousePosDoodlePlane.y * lastMousePosDoodlePlane.y);
691                             var newRadius = Math.sqrt(mousePosDoodlePlane.x * mousePosDoodlePlane.x + mousePosDoodlePlane.y * mousePosDoodlePlane.y);
692                             this.selectedDoodle.radius += (newRadius - oldRadius);
693                             
694                             // Keep within bounds
695                             this.selectedDoodle.radius = this.selectedDoodle.rangeOfRadius.constrain(this.selectedDoodle.radius);
696  						}
697 					}
698 					break;
699 				case ED.Mode.Scale:
700 					if (this.selectedDoodle.isScaleable)
701 					{
702 						// Get sign of scale (negative scales create horizontal and vertical flips)
703 						var signX = this.selectedDoodle.scaleX/Math.abs(this.selectedDoodle.scaleX);
704 						var signY = this.selectedDoodle.scaleY/Math.abs(this.selectedDoodle.scaleY);
705 
706 						// Calculate change in scale (sign change indicates mouse has moved across central axis)
707 						var changeX = mousePosSelectedDoodlePlane.x/lastMousePosSelectedDoodlePlane.x;
708 						var changeY = mousePosSelectedDoodlePlane.y/lastMousePosSelectedDoodlePlane.y;
709 						
710 						// Ensure scale change is same if not squeezable
711 						if (!this.selectedDoodle.isSqueezable)
712 						{
713 							if (changeX > changeY) changeY = changeX;
714 							else changeY = changeX;
715 						}
716 						
717 						// Check that mouse has not moved from one quadrant to another 
718 						if (changeX > 0 && changeY > 0)
719 						{
720 							// Now do scaling
721 							this.selectedDoodle.scaleX = this.selectedDoodle.scaleX * changeX;
722 							this.selectedDoodle.scaleY = this.selectedDoodle.scaleY * changeY;
723 							
724 							// Constrain scale
725 							this.selectedDoodle.scaleX = this.selectedDoodle.rangeOfScale.constrain(Math.abs(this.selectedDoodle.scaleX)) * signX;
726 							this.selectedDoodle.scaleY = this.selectedDoodle.rangeOfScale.constrain(Math.abs(this.selectedDoodle.scaleY)) * signY;
727 						}
728 						else
729 						{
730 							this.mode = ED.Mode.None;
731 						}
732 					}
733 					break;
734 				case ED.Mode.Arc:
735 
736                     // Calculate angles from centre to mouse positions relative to north
737                     var newAngle = this.innerAngle(doodleTop, doodleOrigin, mousePosSelectedDoodlePlane);
738                     var oldAngle = this.innerAngle(doodleTop, doodleOrigin, lastMousePosSelectedDoodlePlane);
739                     
740                     // Work out difference, and sign of rotation correction
741                     var deltaAngle = newAngle - oldAngle;
742                     if (this.selectedDoodle.isArcSymmetrical) deltaAngle = 2 * deltaAngle;
743                     rotationCorrection = 1;
744 
745                     // Arc left or right depending on which handle is dragging
746                     if (this.selectedDoodle.draggingHandleIndex < 2)
747                     {
748                         deltaAngle =  -deltaAngle;
749                         rotationCorrection = -1;
750                     }
751                     
752                     // Clamp to permitted range and stop dragging if exceeded
753                     if (this.selectedDoodle.rangeOfArc.isBelow(this.selectedDoodle.arc + deltaAngle))
754                     {
755                         deltaAngle = this.selectedDoodle.rangeOfArc.min - this.selectedDoodle.arc;
756                         this.selectedDoodle.arc = this.selectedDoodle.rangeOfArc.min;
757                         this.mode = ED.Mode.None;
758                     }
759                     else if (this.selectedDoodle.rangeOfArc.isAbove(this.selectedDoodle.arc + deltaAngle))
760                     {
761                         deltaAngle = this.selectedDoodle.rangeOfArc.max - this.selectedDoodle.arc;
762                         this.selectedDoodle.arc = this.selectedDoodle.rangeOfArc.max;
763                         this.mode = ED.Mode.None;
764                     }
765                     else
766                     {
767                         this.selectedDoodle.arc += deltaAngle;
768                     }
769                     
770                     // Correct rotation with counter-rotation
771                     if (!this.selectedDoodle.isArcSymmetrical)
772                     {
773                         rotationCorrection = rotationCorrection * deltaAngle/2;
774                         this.selectedDoodle.rotation += rotationCorrection;
775                     }
776 
777 					break;
778 				case ED.Mode.Rotate:
779 					if (this.selectedDoodle.isRotatable)
780 					{
781 						// Calculate angles from centre to mouse positions relative to north
782 						var oldAngle = this.innerAngle(doodleTop, doodleOrigin, lastMousePosDoodlePlane);
783 						var newAngle = this.innerAngle(doodleTop, doodleOrigin, mousePosDoodlePlane);
784 						
785 						// Work out difference, and change doodle's angle of rotation by this amount
786 						var deltaAngle = newAngle - oldAngle;
787 						this.selectedDoodle.rotation = this.selectedDoodle.rotation + deltaAngle;
788 					}
789 					break;
790 				case ED.Mode.Apex:
791 					// Move apex to new position
792 					this.selectedDoodle.apexX += (mousePosSelectedDoodlePlane.x - lastMousePosSelectedDoodlePlane.x);
793 					this.selectedDoodle.apexY += (mousePosSelectedDoodlePlane.y - lastMousePosSelectedDoodlePlane.y);
794 					
795 					// Enforce bounds
796 					this.selectedDoodle.apexX = this.selectedDoodle.rangeOfApexX.constrain(this.selectedDoodle.apexX);
797 					this.selectedDoodle.apexY = this.selectedDoodle.rangeOfApexY.constrain(this.selectedDoodle.apexY);
798 					break;
799 				case ED.Mode.Handles:
800 					// Move handles to new position (Stored in a squiggle)
801                     var i = this.selectedDoodle.draggingHandleIndex;
802                     
803                     
804                     // TEMP testing constraining radius
805                     var p = this.selectedDoodle.squiggleArray[0].pointsArray[i];
806                     if (p.length() < this.selectedDoodle.rangeOfRadius.max)
807                     {
808                         this.selectedDoodle.squiggleArray[0].pointsArray[i].x += (mousePosSelectedDoodlePlane.x - lastMousePosSelectedDoodlePlane.x);
809                         this.selectedDoodle.squiggleArray[0].pointsArray[i].y += (mousePosSelectedDoodlePlane.y - lastMousePosSelectedDoodlePlane.y);
810                     }
811                     else
812                     {
813                         if (p.x * p.x > p.y * p.y)
814                         {
815                             var ax = Math.abs(p.x);
816                             p.x = (ax - 1) * p.x/ax;
817                         }
818                         else
819                         {
820                             var ay = Math.abs(p.y);
821                             p.y = (ay - 1) * p.y/ay;
822                         }
823                     }
824 
825                     // Enforce bounds
826 					this.selectedDoodle.squiggleArray[0].pointsArray[i].x = this.selectedDoodle.rangeOfHandlesXArray[i].constrain(this.selectedDoodle.squiggleArray[0].pointsArray[i].x);
827 					this.selectedDoodle.squiggleArray[0].pointsArray[i].y = this.selectedDoodle.rangeOfHandlesYArray[i].constrain(this.selectedDoodle.squiggleArray[0].pointsArray[i].y);
828 					break;
829                 case ED.Mode.Draw:
830                     var p = new ED.Point(mousePosSelectedDoodlePlane.x,mousePosSelectedDoodlePlane.y);
831                     this.selectedDoodle.addPointToSquiggle(p);
832                     break;
833 				default:
834 					break;		
835 			}
836 			            
837 			// Refresh drawing
838 			this.repaint();				
839 		}
840 		
841 		// Store mouse position
842 		this.lastMousePosition = _point;
843 	}
844 }

1700  *
1701  * @class Report
1702  * @property {Canvas} canvas A canvas element used to edit and display the drawing
1703  * @property {Int} breaksInAttached The number of retinal breaks in attached retina
1704  * @property {Int} breaksInDetached The number of retinal breaks in detached retina
1705  * @property {String} largestBreakType The type of the largest retinal break
1706  * @property {Int} largestBreakSize The size in clock hours of the largest retinal break
1707  * @property {Int} lowestBreakPosition The lowest position of any break in clock hours
1708  * @property {String} pvrType The type of PVR
1709  * @property {Int} pvrCClockHours The number of clock hours of posterior PVR type C
1710  * @property {Int} antPvrClockHours The number of clock hours of anterior PVR
1711  * @param Drawing _drawing The drawing object to be analysed
1712  */
1713 ED.Report = function(_drawing)
1714 {
1715    	// Properties
1716 	this.drawing = _drawing;
1717     this.breaksInAttached = 0;
1718     this.breaksInDetached = 0;
1719     this.largestBreakType = 'Not found';
1720     this.largestBreakSize = 0;
1721     this.lowestBreakPosition = 12;
1722     this.pvrType = 'None';
1723     this.pvrCClockHours = 0;
1724     this.antPvrClockHours = 0;
1725     
1726     // Variables
1727     var pvrCDegrees = 0;
1728     var AntPvrDegrees = 0;
1729     var minDegreesFromSix = 180;
1730     
1731     // Create array of doodle classes which are retinal breaks
1732     var breakClassArray = new Array();
1733     breakClassArray["UTear"] = "U tear";
1734     breakClassArray["RoundHole"] = "Round hole";
1735     breakClassArray["Dialysis"] = "Dialysis";
1736     breakClassArray["GRT"] = "GRT";
1737     breakClassArray["MacularHole"] = "Macular hole";
1738     breakClassArray["OuterLeafBreak"] = "Outer Leaf Break";
1739     
1740     // Array of RRD doodles
1741     this.rrdArray = new Array();
1742     
1743     // First iteration to create array of retinal detachments
1744     var i, doodle;
1745 	for (i = 0; i < this.drawing.doodleArray.length; i++)
1746 	{
1747         doodle = this.drawing.doodleArray[i];
1748         
1749         // If its a RRD, add to RRD array
1750         if(doodle.className == "RRD")
1751         {
1752             this.rrdArray.push(doodle);
1753         }
1754     }
1755 
1756     // Second iteration for other doodles
1757 	for (i = 0; i < this.drawing.doodleArray.length; i++)
1758 	{
1759         doodle = this.drawing.doodleArray[i];
1760         
1761         // Star fold - PVR C
1762         if (doodle.className == "StarFold")
1763         {
1764             this.pvrType = 'C';

1766         }
1767         // Anterior PVR
1768         else if (doodle.className == "AntPVR")
1769         {
1770             this.pvrType = 'C';
1771             AntPvrDegrees += doodle.arc * 180/Math.PI;
1772         }
1773         // Retinal breaks
1774         else if (doodle.className in breakClassArray)
1775         {

1779             {
1780                 breakBearing = (Math.round(Math.atan2(doodle.originX, -doodle.originY) * 180/Math.PI) + 360) % 360;
1781             }
1782             else
1783             {
1784                 breakBearing = (Math.round(doodle.rotation * 180/Math.PI + 360)) % 360;
1785             }
1786             
1787             // Bool if break is in detached retina
1788             var inDetached = this.inDetachment(breakBearing);
1789             
1790             // Increment totals

1793                 this.breaksInDetached++;
1794             }
1795             else

1798             }
1799

1801             if (inDetached && doodle.arc > this.largestBreakSize)

1804                 this.largestBreakType = breakClassArray[doodle.className];
1805             }
1806             
1807             // Get lowest break
1808             var degreesFromSix = Math.abs(breakBearing - 180);

1835     var remainder = size % 30;
